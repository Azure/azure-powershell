 [-SubscriptionId <String>] [-DataConnectorCredentials <IDataConnectorCredentialMap[]>]
 [-EnableSystemAssignedIdentity <Boolean?>] [-InstalledSolution <IInstalledSolutionMap[]>]
 [-SkuCapacity <Int32>] [-SkuFamily <String>] [-SkuName <String>] [-SkuSize <String>] [-SkuTier <String>]
 [-Tag <Hashtable>] [-UserAssignedIdentity <String[]>] [-DefaultProfile <PSObject>] [-AsJob] [-NoWait]

 [-UserAssignedIdentity <String[]>] [-DefaultProfile <PSObject>] [-AsJob] [-NoWait] [-Confirm] [-WhatIf]
